# = A team's home page =
# 
# The home page of a team is publicly visible and contains a few interesting
# details about the team.
# 
#     >>> browser.open('http://launchpad.dev/~ubuntu-team')
#     >>> print find_main_content(browser.contents)
#     <...
#     <h1>Ubuntu Team</h1>
#     ...
#     <h2>Contact Details</h2>
#     ...
#     <h2>Team members</h2>
#     ...
#     There are ... direct members of the ... and ... people are members in
#     total...
#     ...
#     <h2>Relationship to other teams</h2>
#     ...is a member of these teams:
#     ...
#     <h2>Your involvement</h2>
#     ...
#     You must ...login...to join or leave this team.
#     ...
# 
# In the above case there's no user logged in, so it doesn't actually show
# what's the user's involvement with the team. If the user logs in, he'll see
# that, though.
# 
#     >>> browser.addHeader('Authorization', 'Basic foo.bar@canonical.com:test')
#     >>> browser.open('http://launchpad.dev/~ubuntu-team')
#     >>> print find_main_content(browser.contents)
#     <...
#     <h2>Your involvement</h2>
#     ...
#     You are a member of this team.
#     ...
# 
# If the logged in user is an indirect member of the team, we'll say that and
# will even show the path from the user to the team.
# 
#     >>> from zope.testbrowser.testing import Browser
#     >>> browser = Browser()
#     >>> browser.addHeader('Authorization', 'Basic test@canonical.com:test')
#     >>> browser.open('http://launchpad.dev/~name18')
#     >>> print find_main_content(browser.contents)
#     <...
#     <h2>Your involvement</h2>
#     ...
#     You are an indirect member of this team:
#     ...
#     ...Sample Person...-&gt;
#     ...Warty Security Team...-&gt;
#     ...Ubuntu Gnome Team...
#     ...
#

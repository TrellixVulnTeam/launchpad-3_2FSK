    # Prints a heading and formatted list of POFiles and latest submissions.
    >>> def print_list_of_contributions(listing):
    ...     listing_head = listing.findAll('thead')[0].findAll('tr')[0]
    ...     heading_cells = listing_head.findAll('th')
    ...     print "%-24s %-10s %-26s %-10s" % (extract_text(heading_cells[0]),
    ...                                        extract_text(heading_cells[1]),
    ...                                        extract_text(heading_cells[2]),
    ...                                        extract_text(heading_cells[3]))
    ...     listing_body = listing.findAll('tbody')[0]
    ...     for row in listing_body.findAll('tr'):
    ...         cells = row.findAll('td')
    ...         product = extract_text(cells[0])
    ...         if len(product) > 24:
    ...             product = product[:21] + '...'
    ...         language = extract_text(cells[1])
    ...         message = extract_text(cells[2]) + ' ' + extract_text(cells[3])
    ...         if len(message) > 26:
    ...             message = message[:23] + '...'
    ...         date = extract_text(cells[4])
    ...         print "%-24s %-10s %-26s %-10s" % (
    ...             product, language, message, date)

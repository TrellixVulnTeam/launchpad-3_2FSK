The bug contacts are now sabdfl and the Launchpad admin team.

  >>> from zope.component import getUtility
  >>> from canonical.launchpad.interfaces import IDistributionSet
  >>> from canonical.launchpad.ftests import login, ANONYMOUS
  >>> login(ANONYMOUS)
  >>> ubuntu = getUtility(IDistributionSet).get(1)
  >>> ubuntu_mozilla_firefox = ubuntu.getSourcePackage("mozilla-firefox")
  >>> sorted([pkg_bug_contact.bugcontact.id for pkg_bug_contact in
  ...         ubuntu_mozilla_firefox.bugcontacts])

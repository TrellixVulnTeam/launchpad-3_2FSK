= Removing an email =

    >>> browser = setupBrowser(auth='Basic foo@baz.com:f')
    >>> browser.open('http://launchpad.dev/~foo/+editemails')
    >>> browser.getControl('baz.foo@canonical.com').selected = True
    >>> browser.getControl('Remove').click()
    >>> for msg in get_feedback_messages(browser.contents):
    ...     print msg
